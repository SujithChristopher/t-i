\begin{equation*}
    \begin{aligned}
        a(t) = \frac{f(t)}{m}
    \end{aligned}
\end{equation*}

\begin{equation*}
    \begin{aligned}
        A(S) = \frac{F(S)}{m}

    \begin{aligned}

\end{equation*}

\begin{equation*}
    \begin{aligned}
        H(S) = \frac{A(S)}{F(S)} = \frac{1}{m}

\end{equation*}
